# Function to create a hazard marker
def create_hazard_marker(ax, x, y, size=1):
    for start_angle, color in zip([0, 90, 180, 270], ['black', 'white', 'black', 'white']):
        wedge = Wedge(center=(x, y), r=size, theta1=start_angle, theta2=start_angle+90, facecolor=color, edgecolor='black')
        ax.add_patch(wedge)

import matplotlib.pyplot as plt
import numpy as np

# Constants
L_fus = 0.74  # Fuselage length in meters
MAC = 0.2  # Mean Aerodynamic Chord in meters
Part = ['Fuselage', 'Engine', 'Tail', 'Wing', 'Battery 1', 'Battery 2', 'Payload']

# Wing leading edge (LEMAC) position in meters (varying as a fraction of fuselage length)
X_LEMAC_values = np.linspace(0.1 * L_fus, 0.8 * L_fus, 50)

# Weights of components (kg) and positions (meters)
W_payload = 7  # kg (Example payload weight)

## Adjusting data to reflect minimum (0 kg) and maximum (7 kg) payloads
min_payload = 0  # kg
max_payload = 7  # kg

# Store results for min and max payloads
Xcg_MAC_min_payload = []
Xcg_MAC_max_payload = []

# Loop through each wing position
for X_LEMAC in X_LEMAC_values:
    # Update wing position
    X[3] = X_LEMAC  # Update the position of the wing
    
    # Min payload
    Weight[-1] = min_payload
    Total_Weight_min = np.sum(Weight)
    X_cg_min = np.sum(Weight * X) / Total_Weight_min  # CG position for min payload
    Xcg_MAC_min_payload.append((X_cg_min - X_LEMAC) / MAC)
    
    # Max payload
    Weight[-1] = max_payload
    Total_Weight_max = np.sum(Weight)
    X_cg_max = np.sum(Weight * X) / Total_Weight_max  # CG position for max payload
    Xcg_MAC_max_payload.append((X_cg_max - X_LEMAC) / MAC)

# Plot results with flipped axes (Xcg/MAC on the x-axis and X_LEMAC/L_fus on the y-axis)
fig, ax = plt.subplots(figsize=(8, 6))

# Plot for min and max payloads
ax.plot(Xcg_MAC_min_payload, X_LEMAC_Lfus_values, label='Min Payload (0 kg)', color='blue', linestyle='--')
ax.plot(Xcg_MAC_max_payload, X_LEMAC_Lfus_values, label='Max Payload (7 kg)', color='red', linestyle='--')

# Add labels, legend, and grid
ax.set_title('CG vs Wing Position (Min and Max Payloads)')
ax.set_xlabel('$X_{cg} / \mathrm{MAC}$ (CG Relative to Wing)')
ax.set_ylabel('$X_{LEMAC} / L_{fus}$ (Wing Leading Edge Relative to Fuselage Length)')

# Constants

SM = 0.05  # Safety margin
CL_alpha_h = 4.703  # Tail lift curve slope in rad^-1
CL_alpha = 8.344  # Aircraft lift curve slope in rad^-1
downwash = 0.85  # Downwash factor (1 - de/dalpha)
l_h_c = 4.0  # Tail arm / MAC ratio
Vh_V_ratio = 0.85  # Velocity ratio (Vh/V)

# COG range (fraction of MAC)
x_cg_values = np.linspace(0.1, 0.6, 100)

# Stability Curve: Eq. 6.2
Sh_S_stability = []
for x_cg in x_cg_values:
    Sh_S = (x_cg - x_ac + SM) / (CL_alpha_h / CL_alpha * downwash * l_h_c * (Vh_V_ratio ** 2))
    Sh_S_stability.append(Sh_S)

# Controllability Curve: Eq. 6.13
Cm_ac = -0.1  # Moment coefficient at the aerodynamic center
# Controllability Curve: Eq. 6.13
Sh_S_controllability = []
for x_cg in x_cg_values:
    Sh_S = -(Cm_ac + (x_cg - x_ac)) / (CL_alpha_h / CL_alpha * l_h_c * (Vh_V_ratio ** 2))
    Sh_S_controllability.append(Sh_S)


# Plotting the Scissor Plot
